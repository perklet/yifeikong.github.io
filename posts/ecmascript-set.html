EcmaScript Set

basic usage

let set = new Set();
let set = new Set([a, b, c]);

set.size  // the size of set
set.add(a)
set.delete(a)
set.clear()
set.has(value)
set.keys()  // the same as keys
set.values()  // return the values as a iterator
