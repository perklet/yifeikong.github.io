C语言中的 vargs 

int max(int n, ...) {
    va_list arg_pointer;
    int result = INT_MIN;

    va_start(arg_pointer, n);
    for (int i = 0; i < n; i++) {
        int arg = va_arg(arg_pointer, int);
        if (arg > result)
            result = arg;
    }
    va_end(arg_pointer);

    return result;
}
